Pengaturan Supabase

Petunjuk Integrasi:

Masukkan URL proyek dan Kunci API Anonim dari dashboard Supabase Anda. Pastikan Anda telah membuat tabel-tabel berikut:

-- 1. Buat Tabel Penduduk
create table penduduk (
  id text primary key,
  created_at timestamp with time zone default now(),
  nik text not null,
  kk text not null,
  nama text not null,
  jk text,
  tempat text,
  tanggal text,
  agama text,
  pendidikan text,
  pekerjaan text,
  "statusKawin" text,
  "statusKeluarga" text,
  rt text,
  rw text,
  dusun text,
  nohp text,
  alamat text
);

-- 2. Buat Tabel Mutasi Desa
create table mutasi_desa (
  id text primary key,
  created_at timestamp with time zone default now(),
  nama text not null,
  jenis text not null,
  tanggal text,
  keterangan text
);
